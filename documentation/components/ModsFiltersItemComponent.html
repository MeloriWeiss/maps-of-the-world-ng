import { ChangeDetectionStrategy, Component, input, signal } from '@angular/core';
import { CollapsibleDirective, SvgComponent } from '@wm/common-ui';

@Component({
  selector: 'wm-mods-filters-item',
  imports: [CollapsibleDirective, SvgComponent],
  templateUrl: './mods-filters-item.component.html',
  styleUrl: './mods-filters-item.component.scss',
  changeDetection: ChangeDetectionStrategy.OnPush,
})
export class ModsFiltersItemComponent {
  filter = input.required<any>();

  isOpened = signal(false);
  isActive = signal(false);

  toggleOpened(event: MouseEvent) {
    event.stopPropagation();

    this.isOpened.set(!this.isOpened());
  }

  toggleActive() {
    this.isActive.set(!this.isActive());
  }
}

@if (filter(); as filter) {
  <button
    (click)="toggleActive()"
    class="sidebar-content-list-btn"
    [class.active]="isActive()"
    [class.opened]="isOpened()"
  >
    @if (filter.extras.length) {
      <button class="open-filter-button" (click)="toggleOpened($event)">
        <svg icon="arrow-tailless" class="svg16"></svg>
      </button>
    }
    <span>{{ filter.name }}</span>
    <span>{{ isOpened() ? '...' : filter.count }}</span>
    @if (filter.extras.length) {
      <div
        wmCollapsible
        [isOpened]="isOpened()"
        class="extra-filters"
      >
        <div class="filters-body-gap"></div>
        @for (item of filter.extras; track item.value) {
          <label class="filter-checkbox">
            <div class="checkbox">
              <input type="checkbox" class="checkbox-input">
              <span class="filter-checkbox-name">{{ item.name }}</span>
            </div>
            <span>{{ item.count }}</span>
          </label>
        }
      </div>
    }
  </button>
}
